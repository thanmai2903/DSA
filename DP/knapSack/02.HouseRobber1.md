[House Robber](https://leetcode.com/problems/house-robber/)

## M-1. Recursion

```cpp
class Solution {
public:
    int rob(vector<int>& nums) {
        return helper(nums, nums.size()-1);
    }

    int helper(vector<int>& a, int i){
        if(i < 0){
            return 0;
        }
        int pick = a[i] + helper(a, i-2);
        int not_pick = helper(a, i-1);
        return max(pick, not_pick);
    }
};
```

## M-2. Memoization

```cpp
class Solution {
public:
    int dp[100+1];
    int rob(vector<int>& nums) {
        memset(dp, -1, sizeof(dp));
        return helper(nums, nums.size()-1);
    }

    int helper(vector<int>& a, int i){
        if(i < 0){
            return 0;
        }
        if(dp[i] != -1){
            return dp[i];
        }
        int pick = a[i] + helper(a, i-2);
        int not_pick = helper(a, i-1);
        return dp[i] = max(pick, not_pick);
    }
};
```

## M-3. tabulation

```cpp
int rob(vector<int>& nums) {
        int n = nums.size();
        if (n == 0) return 0;
        if (n == 1) return nums[0];

        int dp[n];
        dp[0] = nums[0];

        for (int i = 1; i < n; ++i) {
            dp[i] = max(dp[i - 1], nums[i] + (i >= 2 ? dp[i - 2] : 0));
        }

        return dp[n - 1];
    }
```

## M-4. Space Optimization

```cpp
int rob(vector<int>& nums) {
        int n = nums.size();
        int prev2 = 0;
        int prev = nums[0];

        for(int i=1; i<n; ++i){
            int f = prev;
            int s = i>=2 ? prev2+nums[i] : nums[i];
            int curr = max(f,s);
            prev2 = prev;
            prev = curr;
        }
        return prev;
}
```
