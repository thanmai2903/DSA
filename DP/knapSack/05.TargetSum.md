[Target Sum](https://leetcode.com/problems/target-sum/)

## Recursion

```cpp
class Solution {
public:
int res = 0;
    int findTargetSumWays(vector<int>& nums, int target) {
        calculate(nums,0,0,target);
        return res;
    }
    void calculate(vector<int>& nums, int i, int sum, int target){
        if( i == nums.size()){
            if(sum == target){
                res++;
            }
        }else{
            calculate(nums,i+1,sum - nums[i],target);
            calculate(nums,i+1,sum + nums[i],target);
        }
    }
};
```

**Another Way**

```cpp
class Solution {
public:
    int findTargetSumWays(vector<int>& nums, int target) {
        return dp(nums,nums.size()-1,0,target);

    }
    int dp(vector<int>& nums, int i, int sum, int target){
        if( i < 0 && sum == target) return 1;
        if (i < 0){return 0;}

        int plus = dp(nums,i-1,sum + nums[i],target);
        int minus = dp(nums,i-1,sum - nums[i],target);
        return plus + minus;
    }
};
```

## Tabulation

```cpp
class Solution {
public:


    int findTargetSumWays(vector<int>& nums, int target) {
        int sum = 0;
        for(auto num: nums){ sum += num;}
//if net sum = odd then ans is 0 since we can't partition it
        if(abs(target) > sum || (sum + target)%2 != 0) {return 0;}

        int nettarget = (sum+target)/2;
        return countThesubsets(nums,nettarget);
    }

    int countThesubsets(vector<int>& nums,int nettarget){
        //same knapsack idea
        int n = nums.size();
        int dp[n+1][nettarget+1];

        dp[0][0] = 1;
        //if size 0 for any target value then ans = 0
        for(int i=1;i<nettarget+1;i++){dp[0][i] = 0;}

        for(int i=1;i<n+1;i++){
            for(int j=0;j<nettarget+1;j++){
                if(nums[i-1] <= j){// if curr weight is less than capacity then pick ot notpick
                    dp[i][j] = dp[i-1][j] + dp[i-1][j-nums[i-1]];
                }else{//else not pick
                    dp[i][j] = dp[i-1][j];
                }
            }
        }
        return dp[n][nettarget];
    }


};
```
