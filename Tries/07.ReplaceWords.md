[Replace Words](https://leetcode.com/problems/replace-words/)

```cpp
class TrieNode {
public:
    TrieNode *children[26];
    bool isEnd;

    TrieNode(){
        this->isEnd = false;
        for(int i=0; i<26; i++) this->children[i]= NULL;
    }
};
class Solution {
public:
    TrieNode* root;
    Solution() {
        root = new TrieNode();
    }

    void insert(string word){
        TrieNode *temp = root;

        for(auto l : word){
            if(!temp->children[l-'a']) temp->children[l-'a'] = new TrieNode();
            temp = temp->children[l-'a'];
        }

        temp->isEnd = true;
    }
    string check(string word){
        TrieNode *t = root;
        string s = "";

        for(auto i : word){
            if(!t->children[i-'a']) break;
            s += i;
            t = t->children[i-'a'];
            if(t->isEnd) return s;
        }
        return word;
    }
    string replaceWords(vector<string>& dictionary, string sentence) {
        for (auto& str : dictionary) {
            insert(str);
        }
        istringstream ss(sentence);
        string word = "", ans="";

        while( ss>>word){
            ans += check(word);
            ans += ' ';
        }
        ans.pop_back();

        return ans;
    }
};
```
