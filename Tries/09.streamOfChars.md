## Efficient Memory Solution

```cpp
class TrieNode {
public:
    bool isEnd;
    unordered_map<char, TrieNode*> children;
    TrieNode() {
        isEnd = false;
    }
};

class StreamChecker {
    TrieNode* root = new TrieNode();
    deque<char> stream;  // Using a deque to efficiently manage queries.

public:
    void insert(string word) {
        TrieNode* node = root;
        reverse(word.begin(), word.end());
        for (auto ch : word) {
            if (node->children.find(ch) == node->children.end()) {
                node->children[ch] = new TrieNode();
            }
            node = node->children[ch];
        }
        node->isEnd = true;
    }

    bool search(deque<char>& query) {
        TrieNode* node = root;
        for (char ch : query) {
            if (node->children.find(ch) == node->children.end()) {
                return false;
            }
            node = node->children[ch];
            if (node->isEnd)
                return true;
        }
        return false;
    }

    StreamChecker(vector<string>& words) {
        for (auto& w : words) {
            insert(w);
        }
    }

    bool query(char letter) {
        stream.push_front(letter);
        return search(stream);
    }
};
```
