## Que. Delete Middle Element of a Stack

```cpp
#include <iostream>
#include <stack>

using namespace std;

void deleteMid(stack<int>& s, int current, int mid) {
    // Base case: stack is empty or current element is the middle
    if (s.empty() || current == mid) {
        s.pop();
        return;
    }

    int topElement = s.top();
    s.pop();

    // Recursive call to deleteMid for the next element
    deleteMid(s, current + 1, mid);

    // Push the top element back if it's not the middle
    s.push(topElement);
}

void deleteMiddleElement(stack<int>& s) {
    int size = s.size();
    int mid = size / 2;

    // Call the helper function to delete the middle element
    deleteMid(s, 0, mid);
}

int main() {
    stack<int> myStack;

    // Push elements onto the stack
    myStack.push(5);
    myStack.push(2);
    myStack.push(8);
    myStack.push(1);
    myStack.push(3);

    cout << "Original Stack: ";
    while (!myStack.empty()) {
        cout << myStack.top() << " ";
        myStack.pop();
    }
    cout << endl;

    // Reset the stack
    myStack.push(5);
    myStack.push(2);
    myStack.push(8);
    myStack.push(1);
    myStack.push(3);

    // Delete the middle element
    deleteMiddleElement(myStack);

    cout << "Stack after deleting middle element: ";
    while (!myStack.empty()) {
        cout << myStack.top() << " ";
        myStack.pop();
    }
    cout << endl;

    return 0;
}
```
